    get(key: $key) {
      username
      fullName
      workspace {
        name
        drn
        subscriptionType
        iam {
          username
          businessRole
          user {
            fullName
          }
        }
        projects {
          drn
          name
          repositories {
            name
            status
            url
          }
          pipelines {
            pipelineName
            stackId
          }
        }
      }
    }
  }
}
